SequoiaDB 的查询用 json（bson）对象表示，下表以例子的形式显示了 SQL 语句，SequoiaDB shell 语句和 SequoiaDB C 驱动程序语法之间的对照。

| SQL                                            | SequoiaDB shell                                           | C Driver                                 |
| ---------------------------------------------- | --------------------------------------------------------- | ---------------------------------------- |
| insert into bar( a, b ) values( 1, -1 )        | db.foo.bar.insert( { a: 1, b: -1 } )                      | const char *r = "{ a: 1, b: -1 }" ; <br/> jsonToBson ( &obj, r ); // jsonToBson 将一个 json 字符串转换为 bson 对象 <br/> sdbInsert( collection, &obj ); // collection 为集合 bar 的句柄 <br/> |
| select a,b from bar                            | db.foo.bar.find( null, { a: "", b: "" } )                 | const char *r = "{ a: "", b: "" }" ; <br/> jsonToBson ( & select, r ) ; // jsonToBson 将一个 json 字符串转换为 bson 对象 <br/> sdbQuery ( collection, NULL, &select, NULL, NULL, 0, -1, cursor ) ; // collection 为集合 bar 的句柄，cursor 为返回查询结果的句柄 |
| select * from bar                              | db.foo.bar.find()                                         | sdbQuery ( collection, NULL, NULL, NULL, NULL, 0, -1, cursor ) ; // collection 为集合 bar 的句柄，cursor 为返回查询结果的句柄 |
| select * from bar where age = 20               | db.foo.bar.find( { age: 20 } )                            | const char *r = "{ age: 20 }" ; <br/> jsonToBson ( &condition, r ); // jsonToBson 将一个 json 字符串转换为 bson 对象 <br/> sdbQuery ( collection, & condition, NULL, NULL, NULL, 0, -1, cursor ); // collection 为集合 bar 的句柄，cursor 为返回查询结果的句柄 |
| select * from bar where age = 20 order by name | db.foo.bar.find( { age: 20 } ).sort( { name: 1 } )        | const char r1 = "{ age: 20 }" ; <br/> const char r2 = "{ name: 1 }" ; <br/> jsonToBson ( & condition, r1 ) ; // jsonToBson 将一个 json 字符串转换为 bson 对象 <br/> jsonToBson ( &orderBy, r2 ) ; <br/> sdbQuery ( collection, & condition, NULL, & orderBy, NULL, 0, -1, cursor ) ; // collection 为集合 bar 的句柄，cursor 为返回查询结果的句柄 |
| select * from bar where age > 20 and age < 30  | db.foo.bar.find( { age: { $gt: 20, $lt: 30 } } )          | const char *r = "{ age: { $gt: 20, $lt: 30 } }" ; <br/> jsonToBson ( &condition, r ); // jsonToBson 将一个 json 字符串转换为 bson 对象 <br/> sdbQuery ( collection, & condition , NULL, NULL, NULL, 0, -1, cursor ) ; // collection 为集合 bar 的句柄，cursor 为返回查询结果的句柄 |
| create index testIndex on bar( name )          | db.foo.bar.createIndex( "testIndex", { name: 1 }, false ) | const char *r = "{ name: 1 }" ; <br/> jsonToBson ( &obj, r ); // jsonToBson 将一个 json 字符串转换为 bson 对象 <br/> sdbCreateIndex ( collection, &obj, "testIndex", FALSE, FALSE ) // collection 为集合 bar 的句柄 |
| select * from bar limit 20 skip 10             | db.foo.bar.find().limit(20).skip( 10 )                    | sdbQuery ( collection, NULL, NULL, NULL, NULL, 10, 20, cursor ) ; // collection 为集合 bar 的句柄，cursor 为返回查询结果的句柄 |
| select count( * ) from bar where age > 20      | db.foo.bar.find( { age: { $gt: 20 } } ).count()           | const char *r = "{ age: { $gt: 20 } }" ; <br/> jsonToBson ( &condition, r ); // jsonToBson 将一个 json 字符串转换为 bson 对象 <br/> sdbGetCount ( collection, &condition, &count ) ; // collection 为集合 bar 的句柄，count 为返回总数 |
| update bar set a=a+2 where b=-1                | db.foo.bar.update( { $set: { a: 2 } }, { b: -1 } )        | const char r1 = "{ $set: { a: 2 } }" ; <br/> const char r2 = "{ b: -1 }" ; <br/> jsonToBson ( &rule, r1 ) ; // jsonToBson 将一个 json 字符串转换为 bson 对象 <br/> jsonToBson ( &condition, r2 ) ; <br/> sdbUpdate( collection, &rule, &condition, NULL ) ; // collection 为集合 bar 的句柄 |
| delete from bar where a=1                      | db.foo.bar.remove( { a: 1 } )                             | const char *r = "{ a: 1 }" ; <br/> jsonToBson ( &condition, r ) ; // jsonToBson 将一个 json 字符串转换为 bson 对象 <br/> sdbDelete ( collection, &condition, NULL ) ; // collection 为集合 bar 的句柄 |
